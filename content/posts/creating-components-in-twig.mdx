```twig
{% macro Button(attrs = {}) %}
  {% import _self as appIcons %}

  {% set label = '' %}
  {% set title = '' %}
  {% set type = "button" %}
  {% set element = "button" %}
  {% set variant = "primary" %}
  {% set disabled = false %}
  {% set attributes = '' %}
  {% set classname = [] %}

  {% for key, attr in attrs %}
    {% if key == "label" %}
      {% set label =  attr %}
    {% elseif key == "type" %}
      {% set type = attr %}
    {% elseif key == "class" %}
      {% set classname = attr %}
    {% elseif key == "variant" %}
      {% set variant =  attr %}
    {% elseif key == "disabled" %}
      {% set disabled = attr %}
    {% elseif key == "islink" and attr == true %}
      {% set element = "a" %}
    {% else %}
      {% set attributes = attributes ~ key|e('html_attr') ~ '="' ~ attr|e('html_attr') ~ '" ' %}
    {% endif %}
  {% endfor %}

  <{{element}} type="{{ type }}" class=".btn .btn-{{ variant }} {{ classname|join(' ') }}" title="{{label}}" {{attributes|raw}} {% if disabled %} disabled {% endif %} >
    {{ _self.RenderIcon(icon) }}
    {% if label %}
      <span>{{ label }}</span>
    {% endif %}
  </{{element}}>
{% endmacro %}
```
